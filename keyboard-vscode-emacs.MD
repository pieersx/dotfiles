# Comandos de movimiento
- Ctrl + f = Avanzar caracter
- Ctrl + b = Mover hacia atrás caracter
- Ctrl + n = Pasar a la siguiente línea
- Ctrl + p = Ir a la línea anterior
- Ctrl + a = Mover al principio de la línea
- Ctrl + e = Mover al final de la línea

- Ctrl + <- = Avanzar una unidad de palabra
- Alt + b = Retroceder una unidad de palabra
- Ctrl + -> = Retroceder una unidad de palabra
- Alt + f = Avanzar una unidad de palabra

- Ctrl + v = Desplazarse hacia abajo una unidad de pantalla
- ALt + v = Desplazarse hacia arriba una unidad de pantalla

- Ctrl + ⬆️ = Volver al comienzo del párrafo anterior
- Ctrl + ⬇️ = Avanzar al siguiente párrafo final

- Alt + < = Mover a la parte superior del buffer
- Alt + > = Mover a la parte final del buffer

- Alt + g - Alt + g = Saltar a la línea (paleta de comandos)

- Alt + g - n = Saltar al siguiente error
- Alt + g - p = Saltar al error anterior

- Ctrl + l = Centrar la pantalla en la línea actua

- Alt + m = Moverse hasta el primer carácter

# Comandos de búsqueda
- Ctrl + s = Búsqueda incremental hacia adelante
- Ctrl + r = Búsqueda incremental hacia atrás
- Ctrl + Alt - s = Iniciar búsqueda incremental de expresiones regulares
- Ctrl + Alt - r = Iniciar búsqueda incremental de expresiones regulares

- Alt + % = Reemplazar
- Ctrl + Alt + % = Reemplazar con expresiones regular

- Ctrl + Alt + n = Agregar selección a la siguiente coincidencia encontrada
- Ctrl + Alt + n = Agregar selección a la búsqueda anterior

- Alt + s - o = Abra Búsqueda rápida

# Comandos de edición
- Ctrl + d = Borrar derecha
- Ctrl + h = Borrar izquierda
- Alt + \ = Eliminar espacios y tabulaciones alrededor del punto
- Ctrl + x - Ctrl + o = Eliminar líneas en blanco alrededor
- Alt + ^ = Unir dos líneas de forma limpia

- Alt + d = Palabra asesina
- Alt + Backspace = Matar palabra izquierda

- Ctrl + k = Matar desde el cursor actual hasta el final de la línea
- Ctrl + Shit + Backspace = Matar a toda la línea

- Ctrl + w = Matar el texto seleccionado

- Alt + w = Copiar región para matar el anillo
- Ctrl + y = Pegar el último texto que fue cortado

- Ctrl + o = línea ABIERTA

- Ctrl + j = Nueva linea
- Ctrl + m = Nueva linea

- Ctrl + x - h = Selecionar todo

- Ctrl + / = Deshacer
- Ctrl + ; = Activar y desactivar comentarios de línea

- Alt + l = Convertir a minúsculas
- Alt + u = Convertir a mayúsculas
- Alt + c = Convertir a mayúsculas la primera letra de la palabra

# Comandos de marca
- Ctrl + @ = Establezca la marca en el punto y actívela
- Ctrl + Space - Ctrl + Space = Coloque la marca empujándola sobre el anillo de marcación, sin activarla.
- Ctrl + u - Ctrl + Space = Mueva el punto donde estaba la marca y restaure la marca desde el anillo de marcas anteriores.
- Ctrl x - Ctrl + x = Coloque la marca en el punto y actívela; luego mueva el punto a donde solía estar la marca

# Registros de texto
- Ctrl + x - r - s - a = Copiar región en el registro r (copiar a registro)
- Ctrl + x - r - i - a = Insertar texto desde el registro r (insertar-registro)

# Otros comandos
- Ctrl + g = Cancelar
- Ctrl + ' = Sugerencia de IntelliSense
- Ctrl + x = Abrir paleta de comandos
- Ctrl + Alt + Space = Activar o desactivar la visibilidad de la barra lateral
- Ctrl + x - z = Activar o desactivar el modo Zen
- Ctrl + x - Ctrl + c = Cerrar vs code

# Comandos de archivo
- Ctrl + x - Ctrl + f = Apertura rápida de un archivo
- Ctrl + x - Ctrl + s = Guardar
- Ctrl + x - Ctrl + w = Guardar como
- Ctrl + x + s = Guardar todos los archivos
- Ctrl + x - Ctrl + n = Abrir nuevo vs code

# Comandos de manipulación de pestañas y búfer
- Ctrl + x - b = Cambiar a otro buffer abierto
- Ctrl + x - k = Cerrar pestaña actual (buffer)
- Ctrl + x - 0 = Cerrar editores en el grupo actual

- Ctrl + x - 1 = Cerrar editores en otro grupo (dividido).
- Ctrl + x - 2 = Editor dividido horizontalmente
- Ctrl + x - 3 = Editor dividido verticalmente
- Ctrl + x - 4 = Cambiar diseño dividido (vertical a horizontal)

- Ctrl + x - o = Centrar la atención en otro editor dividido
- Ctrl + x - <- Seleccione la pestaña anterior
- Ctrl + x - -> Seleccione la siguiente pestaña

# Argumento de prefijo
- Ctrl + u = argumento universal
- Alt + <number> = argumento dígito
- Alt + - = argumento negativo
